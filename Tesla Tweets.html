def topNHashtags(n):

    pipeline=[
        {"$addFields": {"textArray": {"$split": ["$text", " "]}}},
        {"$unwind": "$textArray"},
        {"$addFields": {"textArray": {"$toLower": "$textArray"}}},
        {"$match": {"textArray": {"$regex":"^#"}}},
        {"$match": {"textArray": {"$nin": ["#tesla\n"]}}},
        {"$group": {
            "_id": "$textArray",
            "frequency": {"$sum":1}
        }},
        {"$sort": {"frequency":-1}},
        {"$limit":n}
    ]
    
    result_list = []
    result = db.tesla_tweets.aggregate(pipeline)
    
    for res in result:
        result_list.append(res)
        
    return result_list

# display top hashtag for top 5 most popular languages

def topNHashtagLanguage(n):
    
    pipeline=[
        {"$addFields": {"textArray": {"$split": ["$text", " "]}}},
        {"$unwind": '$textArray'},
        {"$addFields": {"textArray": {"$toLower": "$textArray"}}},
        {"$match": {"textArray": {"$regex":"^#"}}},
        {"$match": {"textArray": {"$nin": ["#tesla\n", "#..."]}}},
        {"$group": {
        "_id": {"language": "$lang", "hashtag": "$textArray"},
        "frequency": {"$sum":1}
        }},
        {"$sort" : {'frequency' : -1}},
        {"$group": {
        "_id": "$_id.language",
        "total_hashtags": {"$push": {"hashtag": "$_id.hashtag", "frequency": "$frequency"}},
        "total_count" : {"$sum" : 1}
    }},
    {"$sort" : {'total_count' : -1}},
    {"$limit" : 5},
    {"$project": {
        "top5_hashtags": {"$slice": ["$total_hashtags", 0, 5]}}
    }
]
    
    
    result_list = []
    result = db.tesla_tweets.aggregate(pipeline)
    
    for res in result:
        result_list.append(res)
        
    return result_list

def tweetTimes():
    pipeline = [{"$addFields":{'new_date':{ '$dateFromString': {'dateString': '$created_at'}}}},
     {'$addFields':{'hour':{'$hour': '$new_date'}}},
     {'$group':{'_id': '$hour', 'hour_total' : {'$sum' : 1}}},
     {'$sort': {'hour_total' : -1}},
    {'$limit' : 5}
    ]
    result_list = []
    result = db.tesla_tweets.aggregate(pipeline)
    
    for res in result:
        result_list.append(res)
        
    return result_list
